ImPure Function :-


Definition:- An impure function is one that may produce different results even with the same input, or that causes side effects such as modifying external variables, performing I/O operations, etc.

Example:-

function isPrimeWithLog(num) {
if (num <= 1) {
console.log(`${num} is not prime`);
return false;
}
for (let i = 2; i < num; i++) {
if (num % i === 0) {
console.log(`${num} is not prime`);
return false;
}
}
console.log(`${num} is prime`);
return true;
}
console.log(isPrimeWithLog(5)); // Output: "5 is prime", true
console.log(isPrimeWithLog(4)); // Output: "4 is not prime", false
